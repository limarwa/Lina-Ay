        if(!nama || !nomor) return alert("Harap isi Nama dan Nomor Peserta!");

        examData = [...questions].sort(() => Math.random() - 0.5);
        userAnswers = new Array(examData.length).fill(null);
        
        document.getElementById('disp-nama').innerText = nama.toUpperCase();
        document.getElementById('disp-nomor').innerText = nomor;
        document.getElementById('login-screen').classList.add('hidden');

        initNav();
        renderQuestion();
        startTimer();
    };

    function initNav() {
        const grid = document.getElementById('nav-grid');
        grid.innerHTML = '';
        examData.forEach((_, i) => {
            const btn = document.createElement('div');
            btn.className = 'nav-item';
            btn.innerText = i + 1;
            btn.id = `nav-${i}`;
            btn.onclick = () => { currentIndex = i; renderQuestion(); };
            grid.appendChild(btn);
        });
    }

    window.renderQuestion = () => {
        const data = examData[currentIndex];
        document.getElementById('q-number').innerText = `SOAL NO. ${currentIndex + 1}`;
        document.getElementById('q-category').innerText = data.cat;
        document.getElementById('q-text').innerText = data.q;

        const list = document.getElementById('options-list');
        list.innerHTML = '';
        data.options.forEach((opt, i) => {
            const div = document.createElement('div');
            div.className = `option ${userAnswers[currentIndex] === i ? 'selected' : ''}`;
            div.innerHTML = `<div class="opt-letter">${String.fromCharCode(65+i)}</div><div>${opt}</div>`;
            div.onclick = () => { userAnswers[currentIndex] = i; renderQuestion(); };
            list.appendChild(div);
        });

        document.getElementById('prev-btn').disabled = currentIndex === 0;
        document.getElementById('next-btn').innerText = currentIndex === examData.length - 1 ? 'SELESAI' : 'SELANJUTNYA';
        document.getElementById('next-btn').onclick = currentIndex === examData.length - 1 ? openConfirmModal : () => changeQuestion(1);
        updateNav();
    };

    function updateNav() {
        examData.forEach((_, i) => {
            const el = document.getElementById(`nav-${i}`);
            if (!el) return;
            el.className = 'nav-item';
            if(i === currentIndex) el.classList.add('active');
            if(userAnswers[i] !== null) el.classList.add('filled');
        });
    }

    window.changeQuestion = (dir) => {
        currentIndex += dir;
        renderQuestion();
    };

    window.openConfirmModal = () => {
        const remain = userAnswers.filter(a => a === null).length;
        document.getElementById('confirm-msg').innerText = remain > 0 ? `Masih ada ${remain} soal belum dijawab. Yakin ingin mengakhiri?` : "Seluruh soal telah dijawab. Kirim jawaban sekarang?";
        document.getElementById('confirm-modal').classList.replace('hidden', 'flex');
    };

    window.closeConfirmModal = () => document.getElementById('confirm-modal').classList.replace('flex', 'hidden');

    window.finishExam = async () => {
        if (!currentUser) return;
        clearInterval(timerId);
        closeConfirmModal();

        let correct = 0;
        userAnswers.forEach((ans, i) => { if(ans === examData[i].correct) correct++; });
        const score = Math.round((correct / examData.length) * 100);

        try {
            await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'scores'), {
                name: document.getElementById('input-nama').value,
                idNumber: document.getElementById('input-nomor').value,
                score: score,
                timestamp: Date.now(),
                uid: currentUser.uid
            });

            document.getElementById('res-name').innerText = document.getElementById('input-nama').value.toUpperCase();
            document.getElementById('res-id').innerText = document.getElementById('input-nomor').value;
            document.getElementById('final-score').innerText = score;
            document.getElementById('final-stats').innerText = `${correct} / ${examData.length}`;
            document.getElementById('res-timestamp').innerText = new Date().toLocaleString('id-ID', { dateStyle: 'full', timeStyle: 'short' });
            
            renderReview();
            document.getElementById('result-modal').classList.replace('hidden', 'flex');
        } catch (e) { console.error("Error saving score:", e); }
    };

    function renderReview() {
        const list = document.getElementById('review-list');
        list.innerHTML = '';
        examData.forEach((item, idx) => {
            const userAnswerIdx = userAnswers[idx];
            const isCorrect = userAnswerIdx === item.correct;
            const isUnanswered = userAnswerIdx === null;

            const card = document.createElement('div');
            card.className = `p-5 rounded-2xl border-2 transition-all ${isCorrect ? 'bg-green-50 border-green-200' : 'bg-red-50 border-red-200'}`;
            
            let optionsHtml = '';
            item.options.forEach((opt, oIdx) => {
                let statusClass = 'text-slate-500';
                let icon = '';

                if (oIdx === item.correct) {
                    statusClass = 'text-green-700 font-bold';
                    icon = ' ✅';
                } else if (oIdx === userAnswerIdx && !isCorrect) {
                    statusClass = 'text-red-700 font-bold line-through';
                    icon = ' ❌';
                }

                optionsHtml += `<div class="text-sm py-1 border-b border-white/50 last:border-0 ${statusClass}">
                    ${String.fromCharCode(65+oIdx)}. ${opt}${icon}
                </div>`;
            });

            card.innerHTML = `
                <div class="flex justify-between items-start mb-3">
                    <span class="px-2 py-0.5 bg-white rounded text-[10px] font-bold text-slate-400 border uppercase tracking-wider">Nomor ${idx+1}</span>
                    <span class="text-[10px] font-bold ${isCorrect ? 'text-green-600' : 'text-red-600'} uppercase">
                        ${isUnanswered ? 'Tidak Dijawab' : (isCorrect ? 'Benar' : 'Salah')}
                    </span>
                </div>
                <p class="text-slate-800 font-medium mb-4 leading-relaxed">${item.q}</p>
                <div class="space-y-1 bg-white/40 p-3 rounded-xl border border-white/60">
                    ${optionsHtml}
                </div>
            `;
            list.appendChild(card);
        });
    }

    function startTimer() {
        timerId = setInterval(() => {
            timeLeft--;
            const m = Math.floor(timeLeft/60);
            const s = timeLeft%60;
            document.getElementById('timer-display').innerText = `${m}:${s < 10 ? '0'+s : s}`;
            if(timeLeft <= 0) finishExam();
        }, 1000);
    }

    function loadLeaderboard() {
        const q = collection(db, 'artifacts', appId, 'public', 'data', 'scores');
        onSnapshot(q, (snapshot) => {
            const list = document.getElementById('leaderboard-list');
            list.innerHTML = '';
            const scores = snapshot.docs.map(doc => doc.data()).sort((a,b) => b.score - a.score).slice(0, 5);
            if(scores.length === 0) {
                list.innerHTML = '<p class="text-xs italic opacity-50">Belum ada data nilai.</p>';
                return;
            }
            scores.forEach((s, i) => {
                const div = document.createElement('div');
                div.className = "flex justify-between bg-slate-700 p-3 rounded-xl border border-slate-600 items-center";
                div.innerHTML = `
                    <div class="flex items-center gap-3">
                        <span class="w-5 h-5 bg-slate-600 flex items-center justify-center rounded-full text-[10px] font-black">${i+1}</span>
                        <span class="font-bold truncate max-w-[120px] uppercase">${s.name}</span>
                    </div>
                    <span class="text-blue-400 font-black">${s.score}</span>
                `;
                list.appendChild(div);
            });
        }, (err) => {
            console.error("Firestore error:", err);
            document.getElementById('leaderboard-list').innerHTML = '<p class="text-[10px] text-red-400">Gagal memuat skor.</p>';
        });
    }
